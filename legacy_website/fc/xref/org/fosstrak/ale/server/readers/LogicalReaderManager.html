116 	public static String getPropertyValue(String name,  String propertyName) throws NoSuchNameExceptionResponse, SecurityExceptionResponse, ImplementationExceptionResponse {
117 		LogicalReader logRd = logicalReaders.get(name);
118 		List<LRProperty> propList = logRd.getProperties();
119 		Iterator iterator = propList.iterator();
120 		while (iterator.hasNext()) {
121 			LRProperty prop = (LRProperty) iterator.next();
122 			if (prop.getName().equalsIgnoreCase(propertyName)) {
123 				return prop.getValue();
124 			}
125 		}
126 		return null;

157 	 * removes the specified logical readers from the components of the composite reader (see 10.3 API).
158 	 * @param name name of the composite reader
159 	 * @param readers list of readers to be removed
160 	 * @throws NoSuchNameException whenever the specified name is not defined in the logicalReader API
161 	 * @throws InUseException Is thrown when you try to undefine a Reader that is still referenced by EC or CC
162 	 * @throws ImmutableReaderException whenever you want to change a immutable reader
163 	 * @throws SecurityException the operation was not permitted due to access restrictions
164 	 * @throws ImplementationException whenever something goes wrong inside the implementation
165 	 * @throws NonCompositeReaderException addReader or setReader or removeReader was called on a non compositeReader
166 	 */
167 	public static void removeReaders(String name, java.util.List<String> readers) throws NoSuchNameExceptionResponse, InUseExceptionResponse, ImmutableReaderExceptionResponse, NonCompositeReaderExceptionResponse, SecurityExceptionResponse, ImplementationExceptionResponse {
168 		LogicalReader lgRd = logicalReaders.get(name);
169 		if (! (lgRd instanceof CompositeReader)) {
170 			throw new NonCompositeReaderExceptionResponse("reader " + name + " is not composite");
171 		}
172 		// get the readers that are still in the spec
173 		LRSpec spec = lgRd.getLRSpec();
174 		List<String> res = new ArrayList<String>();
175 		if ((spec.getReaders() != null) && (spec.getReaders().getReader().size() > 0)) {
176 			for (String reader : spec.getReaders().getReader()) {
177 				if (!readers.contains(reader)) {
178 					res.add(reader);
179 				}
180 			}
181 		}
182 		
183 		// add the resulting readers
184 		spec.setReaders(new LRSpec.Readers());
185 		spec.getReaders().getReader().addAll(res);
186 		lgRd.update(spec);
187 	}

267 	/**.
268 	 * removes the logicalReader name (see 10.3 API).
269 	 * @param name name for the logical reader to be undefined
270 	 * @throws NoSuchNameException whenever the specified name is not defined in the logicalReader API
271 	 * @throws InUseException Is thrown when you try to undefine a Reader that is still referenced by EC or CC
272 	 * @throws SecurityException the operation was not permitted due to access restrictions
273 	 * @throws ImmutableReaderException whenever you want to change a immutable reader
274 	 * @throws ImplementationException whenever an internal error occurs
275 	 */
276 	public static void undefine(String name) throws NoSuchNameExceptionResponse, InUseExceptionResponse, SecurityExceptionResponse, ImmutableReaderExceptionResponse, ImplementationExceptionResponse {
277 		// the logicalReader must delete himself from its observables
278 		LOG.debug("undefining reader " + name);
279 		LogicalReader reader = LogicalReaderManager.getLogicalReader(name);
280 		
281 		// according to the EPC standard a reader cannot be undefined when there is 
282 		// an active CC or EC pointing to the reader
283 		// this raises an InUseException
284 		if (reader.countObservers() > 0) {
285 			throw new InUseExceptionResponse();
286 		}
287 		
288 		if (reader instanceof CompositeReader) {
289 			CompositeReader composite = (CompositeReader) reader;
290 			composite.unregisterAsObserver();
291 		} else if (reader instanceof BaseReader) {
292 			BaseReader basereader = (BaseReader) reader;
293 			basereader.disconnectReader();
294 			basereader.cleanup();
295 		}
296 		logicalReaders.remove(name);
297 	}

321 	/**
322 	 * creates a new logical Reader according to spec (see 10.3 API). this variant works on jaxb LRSpec
323 	 * @param name name of the new logicalReader
324 	 * @param spec LRSpec how to build the reader
325 	 * @throws DuplicateNameException when a reader name is already defined
326 	 * @throws ValidationException the provided LRSpec is invalid
327 	 * @throws SecurityException the operation was not permitted due to access restrictions
328 	 * @throws ImplementationException whenever something goes wrong inside the implementation 
329 	 */
330 	public static void define(String name, org.fosstrak.ale.server.readers.gen.LRSpec spec) throws DuplicateNameExceptionResponse, ValidationExceptionResponse, SecurityExceptionResponse, ImplementationExceptionResponse {
331 		
332 		LRSpec thespec = new LRSpec();
333 		
334 		// add the readers
335 		thespec.setReaders(new LRSpec.Readers());
336 		if (spec.getReaders() != null) {
337 			thespec.getReaders().getReader().addAll(spec.getReaders());
338 		}
339 		
340 		// set if composite reader or basereader
341 		thespec.setIsComposite(spec.isIsComposite());
342 		
343 		// set the properties 
344 		thespec.setProperties(new LRSpec.Properties());
345 		for (org.fosstrak.ale.server.readers.gen.LRProperty prop : spec.getLRProperty()) {
346 			LRProperty property = new LRProperty();
347 			property.setName(prop.getName());
348 			property.setValue(prop.getValue());
349 			thespec.getProperties().getProperty().add(property);
350 		}
351 		
352 		// at the ReaderType property
353 		LRProperty property = new LRProperty();
354 		property.setName("ReaderType");
355 		property.setValue(spec.getReaderType());
356 		thespec.getProperties().getProperty().add(property);
357 		
358 		define(name, thespec);
359 	}

363 	 * @param name name of the new logicalReader
364 	 * @param spec LRSpec how to build the reader

366 	 * @throws ValidationException the provided LRSpec is invalid
367 	 * @throws SecurityException the operation was not permitted due to access restrictions
368 	 * @throws ImplementationException whenever something goes wrong inside the implementation

370 	public static void define(String name, LRSpec spec) throws DuplicateNameExceptionResponse, ValidationExceptionResponse, SecurityExceptionResponse, ImplementationExceptionResponse {
371 		LogicalReader logRead = LogicalReader.createReader(name, spec);
372 		// establish connection when basereader

376 		logicalReaders.put(name, logRead);

393 	/**
394 	 * This method initializes the manager by loading the definition from the specified file
395 	 * and creating corresponding logical reader stubs.
396 	 * 
397 	 * @param loadFilePath to initialize
398 	 * @throws ImplementationException whenever something goes wrong inside the implementation 
399 	 * @throws SecurityException the operation was not permitted due to access restrictions
400 	 * @throws DuplicateNameException when a reader name is already defined
401 	 * @throws ValidationException the provided LRSpec is invalid
402 	 */
403 	public static void initializeFromFile(String loadFilePath) throws ImplementationExceptionResponse, SecurityExceptionResponse, DuplicateNameExceptionResponse, ValidationExceptionResponse {
404 		
405 		LOG.debug("Initialize LogicalReaderManager");
406 		
407 		// if configuration file path is not set, set it to default value
408 		if (loadFilePath == null) {
409 			loadFilePath = LOAD_FILEPATH;
410 		}
411 		
412 		// try to parse reader configuration file
413 		LOG.debug("Parse configuration file :" + loadFilePath);
414 		List<org.fosstrak.ale.server.readers.gen.LogicalReader> genLogicalReaders;
415 		try {
416 			// initialize jaxb context and unmarshaller
417 			JAXBContext context = JAXBContext.newInstance(JAXB_CONTEXT);
418 			Unmarshaller unmarshaller = context.createUnmarshaller();
419 			unmarshaller.setEventHandler(new javax.xml.bind.helpers.DefaultValidationEventHandler());
420 			
421 			// unmarshal logical reader configuration file
422 			//InputStream inputStream = LogicalReaderManager.class.getResourceAsStream(loadFilePath);
423 			logicalReadersConfiguration = (LogicalReaders) unmarshaller.unmarshal(LogicalReaderManager.class.getResourceAsStream(loadFilePath));
424 			// trying to validate schema
425 			
426 			unmarshaller.setSchema(SCHEMA_FILEPATH);
427 			boolean isValidating = unmarshaller.getSchema()!=null;
428 			
429 			genLogicalReaders = logicalReadersConfiguration.getLogicalReader();
430 		} catch (JAXBException e) {
431 			e.printStackTrace();
432 			return;
433 		}
434 		
435 		// iterate over logical readers
436 		for (org.fosstrak.ale.server.readers.gen.LogicalReader logicalReader : genLogicalReaders) {
437 			
438 			// get logical reader name
439 			String logName = "";
440 			logName = logicalReader.getName();
441 			org.fosstrak.ale.server.readers.gen.LRSpec spec = logicalReader.getLRSpec();
442 			define(logName, spec);	
443 		}
444 			
445 		// set initialized to true
446 		initialized = true;
447 		LOG.debug("LogicalReaderManager successfully initialized");
448 
449 		LOG.debug("starting the readers");
450 		for (LogicalReader reader : LogicalReaderManager.getLogicalReaders()) {
451 			reader.start();
452 		}
453 	}

455 	/**
456 	 * This method stores the current setting of logicalreaders to a .xml file.
457 	 * 
458 	 * @param storeFilePath configurationFilePath to initialize
459 	 * @throws ImplementationException whenever something goes wrong inside the implementation 
460 	 * @throws SecurityException the operation was not permitted due to access restrictions
461 	 * @throws DuplicateNameException when a reader name is already defined
462 	 * @throws ValidationException the provided LRSpec is invalid
463 	 * @throws FileNotFoundException the provided file was not found
464 	 */
465 	public static void storeToFile(String storeFilePath) throws ImplementationExceptionResponse, SecurityExceptionResponse, DuplicateNameExceptionResponse, ValidationExceptionResponse, FileNotFoundException {
466 		
467 		LOG.debug("Store LogicalReaderManager");
468 		
469 		// if store file path is not set, set it to default value
470 		if (storeFilePath == null) {
471 			storeFilePath = STORE_FILEPATH;
472 		}
473 		
474 		// try to generate store file
475 		LOG.debug("Generate store file");
476 		
477 		try {
478 			// initialize jaxb context and marshaller
479 			JAXBContext context = JAXBContext.newInstance(JAXB_CONTEXT);
480 			Marshaller marshaller = context.createMarshaller();
481 
482 			marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, new Boolean(true));
483 			ObjectFactory objFactory = new ObjectFactory();
484 			
485 			LogicalReaders genLogReaders = (LogicalReaders) objFactory.createLogicalReaders();
486 			Iterable<String> names = logicalReaders.keySet();
487 			for (String name : names){
488 				LogicalReader logRd = logicalReaders.get(name);
489 				LRSpec spec = logRd.getLRSpec();
490 				org.fosstrak.ale.server.readers.gen.LogicalReader genLogRd = objFactory.createLogicalReader();
491 				genLogRd.setName(logRd.getName());
492 				genLogRd.setName(name);
493 				org.fosstrak.ale.server.readers.gen.LRSpec genSpec = objFactory.createLRSpec();
494 				genSpec.setIsComposite(spec.isIsComposite());
495 				if (genSpec.isIsComposite()){
496 					if (spec.getReaders() != null) {
497 						Iterator<String> it = spec.getReaders().getReader().iterator();
498 						while(it.hasNext()){
499 							genSpec.getReaders().add(it.next());
500 						}
501 					}
502 				}
503 				else {
504 					Iterator<LRProperty> it = logRd.getProperties().iterator();
505 					while(it.hasNext()){
506 						org.fosstrak.ale.server.readers.gen.LRProperty genProp = objFactory.createLRProperty();
507 						genProp.setName(it.next().getName());
508 						genProp.setValue(it.next().getValue());
509 						genSpec.getLRProperty().add(genProp);
510 					}
511 					genLogRd.setLRSpec(genSpec);
512 					genLogReaders.getLogicalReader().add(genLogRd);
513 				}
514 			}
515 			// store the file to the file path
516 			marshaller.marshal(genLogReaders, new FileOutputStream(storeFilePath));			
517 
518 		} catch (JAXBException e) {
519 			e.printStackTrace();
520 		}
521 		
522 		LOG.info("LogicalReaderManager successfully stored");
523 	}
