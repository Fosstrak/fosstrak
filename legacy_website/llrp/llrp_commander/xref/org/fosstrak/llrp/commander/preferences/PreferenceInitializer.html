22  package org.fosstrak.llrp.commander.preferences;
23  
24  import org.eclipse.core.runtime.preferences.AbstractPreferenceInitializer;
25  import org.eclipse.jface.preference.IPreferenceStore;
26  import org.fosstrak.llrp.client.repository.sql.PostgreSQLRepository;
27  import org.fosstrak.llrp.commander.LLRPPlugin;
28  import org.fosstrak.llrp.commander.ResourceCenter;
29  
30  /**
31   * Class used to initialize default preference values.
32   *
33   * @author Haoning Zhang
34   * @author sawielan
35   * @version 1.0
36   */
37  public class PreferenceInitializer extends AbstractPreferenceInitializer {
38  
39  	/**
40  	 * Initialize the Preference Values
41  	 * 
42  	 * @see org.eclipse.core.runtime.preferences.AbstractPreferenceInitializer#initializeDefaultPreferences()
43  	 */
44  	public void initializeDefaultPreferences() {
45  		IPreferenceStore store = LLRPPlugin.getDefault().getPreferenceStore();
46  		
47  		store.setDefault(PreferenceConstants.P_READER_FROM, "fromXML");
48  		store.setDefault(PreferenceConstants.P_PROJECT, 
49  				ResourceCenter.DEFAULT_ECLIPSE_PROJECT);
50  		store.setDefault(PreferenceConstants.P_STRING, "Default value");		
51  		store.setDefault(PreferenceConstants.P_DEFAULT_EDITOR, 
52  				PreferenceConstants.P_DEFAULT_EDITOR_GRAPHICAL);
53  		store.setDefault(PreferenceConstants.P_WIPE_DB_ON_STARTUP, false);
54  		store.setDefault(PreferenceConstants.P_LOG_RO_ACCESS_REPORTS, false);
55  		store.setDefault(
56  				PreferenceConstants.P_WIPE_RO_ACCESS_REPORTS_ON_STARTUP,
57  				false);
58  		store.setDefault(PreferenceConstants.P_USE_INTERNAL_DB, true);
59  		store.setDefault(PreferenceConstants.P_EXT_DB_IMPLEMENTOR, 
60  				PostgreSQLRepository.class.getName());
61  		store.setDefault(PreferenceConstants.P_EXT_DB_JDBC,
62  				PostgreSQLRepository.JDBC_STR);
63  		store.setDefault(PreferenceConstants.P_EXT_DB_USERNAME, "llrp");
64  		store.setDefault(PreferenceConstants.P_EXT_DB_PWD, "llrp");
65  		
66  	}
67  
68  }
