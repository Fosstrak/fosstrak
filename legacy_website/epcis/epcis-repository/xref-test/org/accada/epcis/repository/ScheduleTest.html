77      public void testNextScheduledHalfHour() throws ImplementationException,
78              SubscriptionControlsException {
79          // scheduled time is every half an hour
80          // always at the top and the bottom of every hour
81          QuerySchedule qs = new QuerySchedule();
82          qs.setSecond("0");
83          qs.setMinute("0,30");
84          Schedule sched = new Schedule(qs);
85  
86          // current time is 1.6.2006 00:00.00
87          GregorianCalendar start = new GregorianCalendar(2006, 5, 1, 0, 0, 0);
88  
89          // get next scheduled time
90          GregorianCalendar act = sched.nextScheduledTime(start);
91  
92          // this is already a valid time!
93          GregorianCalendar exp = (GregorianCalendar) start.clone();
94          assertEquals(exp, act);
95  
96          // add a second to current time
97          // current time is 1.6.2006 00:00.01
98          start.add(Calendar.SECOND, 1);
99  
100         // get next scheduled time
101         act = sched.nextScheduledTime(start);
102 
103         // expected time is 1.6.2006 00:30.00
104         exp = new GregorianCalendar(2006, 5, 1, 0, 30, 0);
105         assertEquals(exp, act);

198     /**
199      * Test for a complex next scheduled time.
200      * 
201      * @throws ImplementationException
202      *             If an error in the implementation occured.
203      * @throws SubscriptionControlsException
204      *             If an error in the scheudle occured.
205      */
206     public void testComplexNextScheduledTime() throws ImplementationException,
207             SubscriptionControlsException {
208         // scheduled time is 1., 10., 20., or 30. of a month,
209         // at 07-11, 13-17, or 20 hours, 50.30 minutes
210         QuerySchedule qs = new QuerySchedule();
211         qs.setSecond("30");
212         qs.setMinute("50");
213         qs.setHour("[7-11],[13-17],20");
214         qs.setDayOfMonth("1,10,20,30");
215         Schedule sched = new Schedule(qs);
216 
217         // current time is 2.1.2010 08:30.00
218         GregorianCalendar start = new GregorianCalendar(2010, 0, 2, 8, 30, 0);
219 
220         // get next scheduled time
221         GregorianCalendar act = sched.nextScheduledTime(start);
222 
223         // expected time is 10.1. 07:50.30
224         GregorianCalendar exp = new GregorianCalendar(2010, 0, 10, 7, 50, 30);
225         assertEquals(exp, act);
226 
227         // add one second to scheduled time
228         start = (GregorianCalendar) exp.clone();
229         start.add(Calendar.SECOND, 1);
230 
231         // get next scheduled time
232         act = sched.nextScheduledTime(start);
233 
234         // expected time is 10.1. 08:50.30
235         exp = new GregorianCalendar(2010, 0, 10, 8, 50, 30);
236         assertEquals(exp, act);
237 
238         // add some more time to scheduled time
239         start = (GregorianCalendar) exp.clone();
240         start.add(Calendar.HOUR, 3);
241         start.add(Calendar.SECOND, 1);
242 
243         // get next scheduled time
244         act = sched.nextScheduledTime(start);
245 
246         // expected time is 10.1. 13:50.30
247         exp = new GregorianCalendar(2010, 0, 10, 13, 50, 30);
248         assertEquals(exp, act);
249 
250         // add some more time to scheduled time
251         start = (GregorianCalendar) exp.clone();
252         start.add(Calendar.DAY_OF_MONTH, 1);
253 
254         // get next scheduled time
255         act = sched.nextScheduledTime(start);
256 
257         // expected time is 20.1. 07:50.30
258         exp = new GregorianCalendar(2010, 0, 20, 07, 50, 30);
259         assertEquals(exp, act);
260     }

310     /**
311      * Prints the date and time of the given Calendar.
312      * 
313      * @param cal
314      *            The calendar to print.
315      */
316     private void printDateTime(final GregorianCalendar cal) {
317         System.out.println("Year: "
318                 + cal.get(Calendar.YEAR)
319                 // Calendar starts months with 0
320                 + ", Month: " + (cal.get(Calendar.MONTH) + 1) + ", Day: "
321                 + cal.get(Calendar.DAY_OF_MONTH) + ", Hour: "
322                 + cal.get(Calendar.HOUR_OF_DAY) + ", Minute: "
323                 + cal.get(Calendar.MINUTE) + ", Second: "
324                 + cal.get(Calendar.SECOND)
325                 // Calendar week days start on sunday
326                 + ", Weekday: " + (cal.get(Calendar.DAY_OF_WEEK) - 1));
327     }
