33  /**
34   * A simple test utility class for demonstrating how to send masterdata queries
35   * to the Fosstrak EPCIS query module.
36   * 
37   * @author Marco Steybe
38   */
39  public class SimpleMasterDataQueryTest {
40      
41      // Note: keep the methods in this class static in order to prevent them from
42      // being executed when building the project with Maven.
43  
44      public static void main(String[] args) throws Exception {
45          // configure the query service
46          String queryUrl = QueryClientTestHelper.LOCAL_EPCIS_QUERY_URL;
47          QueryControlClient client = new QueryControlClient();
48          client.configureService(new URL(queryUrl), null);
49  
50          // create a query in its XML form and send it to the repository
51          InputStream xmlStream = QueryClientTestHelper.getInputStream(QueryClientTestHelper.SAMPLE_MASTERDATA_QUERY_XML);
52          System.out.println("sending query request (" + xmlStream.available() + " bytes) ...");
53          QueryResults results = client.poll(xmlStream);
54          // print the results to System.out
55          QueryResultsParser.queryResultsToXml(results, System.out);
56          
57          // create a query Poll object and send it to the query service
58          Poll poll = createPoll();
59          System.out.println("sending query ...");
60          results = client.poll(poll);
61          QueryResultsParser.queryResultsToXml(results, System.out);
62      }
63  
64      /**
65       * Creates and returns a simple EPCIS query Poll object.
66       */
67      private static Poll createPoll() {
68          // construct the query parameters
69          QueryParam queryParam1 = new QueryParam();
70          queryParam1.setName("includeAttributes");
71          queryParam1.setValue("true");
72  
73          QueryParam queryParam2 = new QueryParam();
74          queryParam2.setName("includeChildren");
75          queryParam2.setValue("true");
76  
77          QueryParam queryParam3 = new QueryParam();
78          queryParam3.setName("EQ_name");
79          ArrayOfString queryParamValue3 = new ArrayOfString();
80          queryParamValue3.getString().add("urn:epc:id:sgln:0614141.00729.shipping");
81          queryParam3.setValue(queryParamValue3);
82  
83          // add the query parameters to the list of parameters
84          QueryParams queryParams = new QueryParams();
85          queryParams.getParam().add(queryParam1);
86          queryParams.getParam().add(queryParam2);
87          queryParams.getParam().add(queryParam3);
88  
89          // create the Poll object
90          Poll poll = new Poll();
91          poll.setQueryName("SimpleMasterDataQuery");
92          poll.setParams(queryParams);
93          return poll;
94      }
