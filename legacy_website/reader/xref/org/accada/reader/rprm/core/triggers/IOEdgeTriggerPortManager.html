21  package org.accada.reader.rprm.core.triggers;
22  
23  import java.util.Enumeration;
24  
25  import org.accada.reader.rprm.core.Trigger;
26  
27  /**
28   * This class represents the io edge trigger manager and implements some
29   * important methods to notify the triggers.
30   * @author Markus Vitalini
31   */
32  public abstract class IOEdgeTriggerPortManager extends IOTriggerPortManager {
33  
34     /**
35      * Notifies the triggers that an event was received. This method checks the
36      * event and if necessary the corresponding trigger is fired.
37      * @param recPin
38      *           The received pin
39      * @param recRising
40      *           The flag if the signal is falling or rising
41      */
42     final void checkAndNotifyTriggers(final String recPin, 
43           final boolean recRising) {
44  
45        Enumeration iterator = super.getTriggers().elements();
46        Trigger cur;
47  
48        while (iterator.hasMoreElements()) {
49           cur = (Trigger) iterator.nextElement();
50  
51           // type=<rising|falling>;port=<n>;pin=<n>
52  
53           String pin;
54           boolean rising;
55  
56           if (cur.getValue().indexOf("rising") < 0) {
57              rising = false;
58           } else {
59              rising = true;
60           }
61           int second = cur.getValue().lastIndexOf(';');
62           final int offset = 5;
63           pin = cur.getValue().substring(second + offset);
64  
65           if (validate(recPin, recRising, pin, rising)) {
66              cur.fire();
67              System.out.println("fire " + cur.getName());
68           }
69  
70        }
71  
72     }
73  
74     /**
75      * Validate the values of the edge trigger.
76      * @param recPin
77      *           The received pin
78      * @param recRising
79      *           The receive rising flag
80      * @param pin
81      *           The pin of the trigger
82      * @param rising
83      *           The rising flag of the trigger
84      * @return 'ture' if the values matches, 'false' otherwise
85      */
86     final boolean validate(final String recPin, final boolean recRising,
87           final String pin, final boolean rising) {
88  
89        return (recPin.equals(pin)) && (recRising == rising);
90  
91     }
92  
93  }
