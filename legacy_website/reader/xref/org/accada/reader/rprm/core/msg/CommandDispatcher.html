134 /**
135  * The command dispatcher dispatches the commands by invoking the methods on the
136  * ReaderDevice and creates the response. The commands and replies don't depend
137  * on a specific message format the <code>CommandDispatcher</code> uses an
138  * intermediate representation (IR) of the commands and responses. As the IR
139  * tree we use the classes generated by JAXB. An XML message is mapped
140  * automatically into such an IR so get there the IR for free and don't need
141  * another level of indirection (which would deteriorate performance).<br>
142  * Other message formats (e.g., the text message format) are required to parse
143  * the message into the JAXB classes IR before invoking the command dispatcher.
144  * <br>
145  * <br>
146  * Because every EPC reader has only one instance of a
147  * <code>CommandDispatcher</code> this is a singleton class.
148  * 
149  * @author Andreas Fürer, ETH Zurich Switzerland, Winter 2005/06
150  * 
151  */

205 	/**
206 	 * Dispatches a <code>Command</code>.
207 	 * 
208 	 * @param command
209 	 *            The <code>Command</code> to execute.
210 	 * @param im
211 	 *            The <code>IncomingMessage</code> that caused this command
212 	 *            invocation.
213 	 * @return The <code>Reply</code> of the command execution or a
214 	 *         <code>null</code> reply if anything went wrong.
215 	 */
216 	public Reply dispatch(final Command command, final IncomingMessage im)
217 	/* throws EventLayerException */{
218 		try {
219 			Reply reply = replyFactory.createReply();
220 
221 			log.debug("dispatch command");
222 
223 			if (command.getReaderDevice() != null) {
224 				ReaderDeviceCommand rdCommand = command.getReaderDevice();
225 				ReaderDeviceReply rdReply = executeCommand(rdCommand, command
226 						.getTargetName(), im);
227 				reply.setReaderDevice(rdReply);
228 			}
229 
230 			else if (command.getSource() != null) {
231 				SourceCommand srcCommand = command.getSource();
232 				SourceReply srcReply = executeCommand(srcCommand, command
233 						.getTargetName(), im);
234 				reply.setSource(srcReply);
235 			}
236 
237 			else if (command.getReadPoint() != null) {
238 				ReadPointCommand rdpCommand = command.getReadPoint();
239 				ReadPointReply rpReply = executeCommand(rdpCommand, command
240 						.getTargetName(), im);
241 				reply.setReadPoint(rpReply);
242 			}
243 
244 			else if (command.getTrigger() != null) {
245 				TriggerCommand tgCommand = command.getTrigger();
246 				TriggerReply tgReply = executeCommand(tgCommand, command
247 						.getTargetName(), im);
248 				reply.setTrigger(tgReply);
249 			}
250 
251 			else if (command.getTagSelector() != null) {
252 				TagSelectorCommand tsCommand = command.getTagSelector();
253 				TagSelectorReply tsReply = executeCommand(tsCommand, command
254 						.getTargetName(), im);
255 				reply.setTagSelector(tsReply);
256 			}
257 
258 			else if (command.getNotificationChannel() != null) {
259 				NotificationChannelCommand notificationCommand = command
260 						.getNotificationChannel();
261 				NotificationChannelReply notReply = executeCommand(
262 						notificationCommand, command.getTargetName(), im);
263 				reply.setNotificationChannel(notReply);
264 			}
265 
266 			else if (command.getDataSelector() != null) {
267 				DataSelectorCommand dsCommand = command.getDataSelector();
268 				DataSelectorReply dsReply = executeCommand(dsCommand, command
269 						.getTargetName(), im);
270 				reply.setDataSelector(dsReply);
271 			}
272 
273 			else if (command.getEventType() != null) {
274 				EventTypeCommand etCommand = command.getEventType();
275 				EventTypeReply etReply = executeCommand(etCommand, command
276 						.getTargetName(), im);
277 				reply.setEventType(etReply);
278 			}
279 
280 			else if (command.getTriggerType() != null) {
281 				TriggerTypeCommand ttCommand = command.getTriggerType();
282 				TriggerTypeReply ttReply = executeCommand(ttCommand, command
283 						.getTargetName(), im);
284 				reply.setTriggerType(ttReply);
285 			}
286 
287 			else if (command.getFieldName() != null) {
288 				FieldNameCommand fnCommand = command.getFieldName();
289 				FieldNameReply fnReply = executeCommand(fnCommand, command
290 						.getTargetName(), im);
291 				reply.setFieldName(fnReply);
292 			}
293 
294 			else if (command.getTagField() != null) {
295 				TagFieldCommand tfCommand = command.getTagField();
296 				TagFieldReply tfReply = executeCommand(tfCommand, command
297 						.getTargetName(), im);
298 				reply.setTagField(tfReply);
299 			}
300 
301 			// TODO: Exception werfen, wenn kein Kommando ausgeführt??
302 
303 			/* If no exception is thrown, everything went well :-) */
304 			reply.setId(command.getId());
305 			/* TODO: Typ in Schema ändern auf HexStringReturn Type */
306 			reply.setResultCode(MessagingConstants.NO_ERROR);
307 			return reply;
308 		} catch (ReaderProtocolException e) {
309 			Reply errorReply = handleReaderProtocolException(e, command.getId());
310 			return errorReply;
311 		} catch (Exception e) {
312 			log.error("Unexpected exception while dispatching commands: " + e);
313 			Reply errorReply = handleReaderProtocolException(command.getId(),
314 					MessagingConstants.ERROR_UNKNOWN,
315 					MessagingConstants.ERROR_UNKNOWN_STR, e.getMessage());
316 			return errorReply;
317 		}
318 	}
