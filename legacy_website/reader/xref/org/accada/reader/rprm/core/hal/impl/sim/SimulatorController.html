165 	public SimulatorController(String halName) {
166 		this.halName = halName;
167 		this.initialize();
168 		
169 		log.debug("Simulator: " + simType);
170 		reader = new HashMap();
171 		
172 		for (int i = 0; i < nOfReadPoints; i++){
173 			reader.put( readPointNames[i] , new HashSet() );
174 		}
175 		
176 		
177 		try{
178 			//Dynamic class loading
179 			//1.) The full package path 
180 			Class clazz = Class.forName(simType);
181 			//2.) State the types of the constructors arguments. here: only one argument, of type String
182 			java.lang.reflect.Constructor co = clazz.getConstructor(new Class[] {});
183 			//3.) Create new instance with the argument's values
184 			simulator = (SimulatorEngine) co.newInstance(null);
185 			simulator.initialize(this);	
186 		}
187 		catch(ClassNotFoundException cnfe){
188 			cnfe.printStackTrace();
189 			System.exit(1);
190 		}
191 		catch(Exception e){
192 			e.printStackTrace();
193 			System.exit(1);
194 		}		
195 	}

197 	/**
198 	 * Initialize a reader.
199 	 */
200 	public void initialize(){
201 		
202 		this.props = new ControllerProperties(halName);
203 		
204 		//sets the parameters according to the properties file
205 		try{
206 			frequency = Long.parseLong(props.getParameter("frequency"));
207 			
208 			simType = props.getParameter("simType");
209 			
210 			nOfReadPoints = Integer.parseInt(props.getParameter("numberOfReadPoints"));
211 			readPointNames = new String[nOfReadPoints];
212 			readyReadPoints = new Vector<String>();
213 			for (int i = 0; i < nOfReadPoints; i++){
214 				readPointNames[i] = this.props.getParameter("readPoint_" + (i + 1));
215 				readyReadPoints.add(readPointNames[i]);
216 			}
217 		}	
218 		catch (Exception e){
219 			String message="Couldn't initialize the reader:" + e.getMessage();
220 			log.debug(message);
221 			e.printStackTrace();
222 		}
223 	}

333 	 * @see org.accada.reader.hal.HardwareAbstraction#identify(java.lang.String[])
334 	 */
335 	public Observation[] identify(String[] readPointNames)
336 			throws HardwareException {
337 		// Each read point gets its own Observation
338 		Observation[] observations = new Observation[readPointNames.length];
339 		
340 		for (int i = 0; i < readPointNames.length; i++){
341 			
342 			observations[i] = new Observation();
343 			observations[i].setHalName(this.halName);
344 			observations[i].setReadPointName(readPointNames[i]);
345 			
346 			//Get the tag IDs on this antenna
347 			Vector<String> v = new Vector<String>();
348 			Iterator it = ((HashSet)reader.get(readPointNames[i])).iterator();
349 			while (it.hasNext()){
350 				Tag t = (Tag) it.next();
351 				v.add(t.getTagID());
352 			}
353 			
354 			observations[i].setTagIds(v);
355 			observations[i].setTimestamp(System.currentTimeMillis());
356 			
357 			if (!readyReadPoints.contains(readPointNames[i])) {
358 				log.error("Read point \"" + readPointNames[i] + "\" is not ready.");
359 				observations[i].setTagIds(new Vector());
360 				observations[i].successful = false;
361 			} else if (continuousIdentifyErrors.contains(readPointNames[i])) {
362 				observations[i].successful = false;
363 			} else if (identifyError.contains(readPointNames[i])) {
364 				observations[i].successful = false;
365 				identifyError.remove(readPointNames[i]);
366 			} else {
367 				observations[i].successful = true;
368 			}
369 			
370 			log.debug(observations[i].toString());
371 		}
372 		return observations;
373 	}

407 	 * @see org.accada.reader.hal.HardwareAbstraction#readBytes(java.lang.String, java.lang.String, int, int, int, java.lang.String[])
408 	 */
409 	public byte[] readBytes(String readPointName, String id, int memoryBank,
410 			int offset, int length, String[] passwords)
411 			throws HardwareException, UnsupportedOperationException {
412 		if (readyReadPoints.contains(readPointName)) {
413 			
414 			if (continuousReadErrors.contains(readPointName)) {
415 				throw new HardwareException(HardwareException.SERVICECODE_READ, readPointName, halName);
416 			} else if (readError.contains(readPointName)) {
417 				readError.remove(readPointName);
418 				throw new HardwareException(HardwareException.SERVICECODE_READ, readPointName, halName);
419 			}
420 			
421 			byte[] byteData;
422 			Tag tag = null;
423 			if (contains(readPointName, id)) {
424 				Iterator it = ((HashSet)reader.get(readPointName)).iterator();
425 				while (it.hasNext()){
426 					Tag curTag = (Tag)it.next();
427 					if (curTag.getTagID().equalsIgnoreCase(id)) {
428 						tag = curTag;
429 						break;
430 					}
431 				}
432 			}
433 			
434 			if(tag != null){
435 				byteData = tag.readData(memoryBank, offset, length);
436 			}
437 			else{
438 				String message="Specified tag not in range";
439 				throw new HardwareException(HardwareException.SERVICECODE_READ, readPointName, halName, message);	
440 			}
441 			
442 			return byteData;
443 		} else {
444 			throw new HardwareException(HardwareException.SERVICECODE_INITIALIZE, readPointName, halName, "Read point is not ready.");
445 		}
446 	}
