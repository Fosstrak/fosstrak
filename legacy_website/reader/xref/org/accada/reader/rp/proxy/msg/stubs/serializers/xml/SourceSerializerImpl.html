483 	/*
484 	 * (non-Javadoc)
485 	 * 
486 	 * @see org.accada.reader.testclient.command.SourceSerializer#read(org.accada.reader.testclient.command.DataSelectorSerializer,
487 	 *      java.util.Hashtable)
488 	 */
489 	public String read(DataSelector dataSelector, Hashtable passwords) {
490 		resetCommand();
491 		SourceCommand.Read r = cmdFactory.createSourceCommandRead();
492 
493 		// Set DataSelector if optional parameter is available
494 		if (dataSelector != null) {
495 			r.setDataSelector(dataSelector.getName());
496 		}
497 
498 		// TODO: @Markus: passwords sollte keine Hashtable sein
499 		// Convert Hashtable into String[]
500 		String[] strPasswords;
501 		if (passwords != null) {
502 			strPasswords = new String[passwords.size()];
503 			Iterator it = passwords.values().iterator();
504 			int index = 0;
505 			while (it.hasNext()) {
506 				String s = (String) it.next();
507 				strPasswords[index] = s;
508 				index++;
509 			}
510 		} else {
511 			strPasswords = new String[0];
512 		}
513 
514 		// Set passwords if optional parameter is available
515 		if (passwords != null) {
516 			HexStringListType.List list = cmdFactory
517 					.createHexStringListTypeList();
518 			HexStringListType listType = cmdFactory
519 					.createHexStringListType();
520 			list.getValue().addAll(toHexStringList(strPasswords));
521 			listType.setList(list);
522 			r.setPasswords(listType);
523 		}
524 
525 		srcCommand.setRead(r);
526 		return serializeCommand();
527 	}
