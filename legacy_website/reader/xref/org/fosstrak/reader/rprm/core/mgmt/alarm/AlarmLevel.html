23  /**
24   * The level of a reported alarm. When the host requests an alarm it defines the
25   * alarm level (see <code>AlarmControl.setLevel</code>). This additional
26   * information is then returned by the reader along with each occurrence of that
27   * alarm. It is expected that some management systems may choose to filter based
28   * on the alarm level. The alarm levels are modeled after The syslog Protocol
29   * IETF draft.
30   */
31  public enum AlarmLevel {
32  	
33  	/*
34  	 * System is unusable
35  	 */
36  	EMERGENCY,
37  	
38  	/*
39  	 * Action must be taken immediately
40  	 */
41  	ALERT,
42  	
43  	/*
44  	 * Critical conditions
45  	 */
46  	CRITICAL,
47  	
48  	/*
49  	 * Error conditions
50  	 */
51  	ERROR,
52  	
53  	/*
54  	 * Warning conditions
55  	 */
56  	WARNING,
57  	
58  	/*
59  	 * Normal but significant conditions
60  	 */
61  	NOTICE,
62  	
63  	/*
64  	 * Informational messages
65  	 */
66  	INFORMATIONAL,
67  	
68  	/*
69  	 * Debug level messages
70  	 */
71  	DEBUG;
72  	
73  	
74  	/**
75  	 * Converts this <code>AlarmLevel</code> instance to an <code>int</code>
76  	 * value.
77  	 * 
78  	 * @return <code>int</code> representation of this <code>AlarmLevel</code>
79  	 *         instance
80  	 */
81  	public final int toInt() {
82  		AlarmLevel[] values = AlarmLevel.values();
83  		for (int index = 0; index < values.length; index++) {
84  			if (values[index] == this) return index;
85  		}
86  		return -1;
87  	}
88  	
89  	/**
90  	 * Converts an <code>int</code> value to an
91  	 * <code>AlarmLevel instance</code>. If the <code>int</code> value is
92  	 * out of range it returns <code>null</code>.
93  	 * 
94  	 * @param i
95  	 *            The <code>int</code> value
96  	 * @return The <code>AlarmLevel</code> representation of the
97  	 *         <code>int</code> value
98  	 */
99  	public static final AlarmLevel intToEnum(final int i) {
100 		try {
101 			return AlarmLevel.values()[i];
102 		} catch (ArrayIndexOutOfBoundsException e) {
103 			return null;
104 		}
105 	}
