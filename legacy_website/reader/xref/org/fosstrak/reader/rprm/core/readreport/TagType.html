145    /**
146     * Adds events to the tagEvent list.
147     * @param tagEvents
148     *           The events to add
149     */
150    public final void addTagEvents(final Hashtable tagEvents) {
151 
152       Enumeration iterator = tagEvents.elements();
153       TagEventType cur;
154       while (iterator.hasMoreElements()) {
155          cur = (TagEventType) iterator.nextElement();
156          this.tagEvents.put(cur.getEventType(), cur);
157       }
158 
159    }

199    /**
200     * Checks if a tag field is in the tagfield list of this tag.
201     * @param name
202     *           The name of the tagfield to check
203     * @return Returns 'true' if a TagField is available, 'false' otherwise
204     */
205    public final boolean containsTagField(final String name) {
206 
207       Enumeration iterator = tagFields.elements();
208       TagFieldValueParamType curTagField;
209       while (iterator.hasMoreElements()) {
210          curTagField = (TagFieldValueParamType) iterator.nextElement();
211          if (curTagField.getTagFieldName().equals(name)) {
212             return true;
213          }
214       }
215       return false;
216 
217    }
