21  package org.fosstrak.reader.rprm.core.msg.util;
22  
23  /**
24   * A pool of threads.
25   * <p>
26   * Avoids the expense of thread creation by pooling threads. The threads are recycled 
27   * after the job is completed.
28   * If the maximum pool size is reached, jobs wait for a free thread.
29   * 
30   * @author Andreas Fürer, ETH Zurich Switzerland, Winter 2005/06
31   *
32   */
33  
34  public class ThreadPool extends Object {
35  	private BlockingQueue idleWorkers;
36  	
37  	private ThreadPoolWorker[] workerList;
38  	
39  	public ThreadPool(int numberOfThreads) {
40  		// make sure that its at least one
41  		numberOfThreads = Math.max(1, numberOfThreads);
42  		
43  		idleWorkers = new BlockingQueue(numberOfThreads);
44  		workerList = new ThreadPoolWorker[numberOfThreads];
45  		
46  		for (int i = 0; i < workerList.length; i++) {
47  			workerList[i] = new ThreadPoolWorker(idleWorkers);
48  		}
49  	}
50  	
51  	public void execute(Runnable target) throws InterruptedException {
52  		// block (forever) until a worker is available
53  		ThreadPoolWorker worker = (ThreadPoolWorker) idleWorkers.pop();
54  		worker.process(target);
55  	}
56  	
57  	public void stopRequestIdleWorkers() {
58  		try {
59  			int size = idleWorkers.size();
60  			for (int i = 0; i < size; i++) {
61  				((ThreadPoolWorker) idleWorkers.pop()).stopRequest();
62  			}
63  		} catch (InterruptedException x) {
64  			Thread.currentThread().interrupt(); // re-assert
65  		}
66  	}
67  	
68  	public void stopRequestAllWorkers() {
69  		// Stop the idle ones first
70  		// productive.
71  		stopRequestIdleWorkers();
72  		
73  		// give the idle workers a quick chance to die
74  		try {
75  			Thread.sleep(250);
76  		} catch (InterruptedException x) {
77  		}
78  		
79  		// Step through the list of ALL workers.
80  		for (int i = 0; i < workerList.length; i++) {
81  			if (workerList[i].isAlive()) {
82  				workerList[i].stopRequest();
83  			}
84  		}
85  	}
86  }
