145 	/**
146 	 * Validates the SenderHandshakeMessage
147 	 * 
148 	 * @return true if message is valid, false otherwise
149 	 */
150 	public boolean isValid() {
151 		/* check the lengths */
152 		if (senderSignature.length() != LENGTH_SENDER_SIGNATURE) {
153 			return false;
154 		}
155 		if (specVersionRequest.length() != LENGTH_SPEC_VERSION_REQUEST) {
156 			return false;
157 		}
158 		if (senderFormatRequest == null
159 				|| senderFormatRequest.length() != LENGTH_SENDER_FORMAT_REQUEST) {
160 			return false;
161 		}
162 
163 		if (receiverFormatRequest == null) {
164 			/* if not set use the message type of the request */
165 			receiverFormatRequest = senderFormatRequest;
166 		}
167 
168 		if (receiverFormatRequest.length() != LENGTH_RECEIVER_FORMAT_REQUEST) {
169 			return false;
170 		}
171 
172 		/* validate contents */
173 		if (!senderSignature.equals(SENDER_SIGNATURE)) {
174 			return false;
175 		}
176 		if (!specVersionRequest.equals(SPEC_VERSION_REQUEST)) {
177 			return false;
178 		}
179 		if (!senderFormatRequest.equals(SENDER_FORMAT_REQUEST_TEXT)
180 				&& !senderFormatRequest.equals(SENDER_FORMAT_REQUEST_XML)) {
181 			return false;
182 		}
183 		if (!receiverFormatRequest.equals(RECEIVER_FORMAT_REQUEST_TEXT)
184 				&& !receiverFormatRequest.equals(RECEIVER_FORMAT_REQUEST_XML)) {
185 			return false;
186 		}
187 
188 		return true;
189 	}

216 		} else if (getReceiverFormatRequest().equals(

218 			return MessageFormat.TEXT;
219 		} else {
220 			return null;
221 		}
222 	}
