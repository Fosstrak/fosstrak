116 	public String serialize(Reply r) throws MessageSerializingException {
117 		StringWriter writer = new StringWriter();
118 		writer.write(serializeReplyHeader(r));
119 		if (r.getReaderDevice() != null) {
120 			ReaderDeviceReply rdReply = r.getReaderDevice();
121 			writer.write(serializeReaderDevice(rdReply));
122 		} else if (r.getSource() != null) {
123 			SourceReply srcReply = r.getSource();
124 			writer.write(serializeSource(srcReply));
125 		} else if (r.getReadPoint() != null) {
126 			ReadPointReply rpReply = r.getReadPoint();
127 			writer.write(serializeReadPoint(rpReply));
128 		} else if (r.getTagSelector() != null) {
129 			TagSelectorReply tsReply = r.getTagSelector();
130 			writer.write(serializeTagSelector(tsReply));
131 		} else if (r.getDataSelector() != null) {
132 			DataSelectorReply dsReply = r.getDataSelector();
133 			writer.write(serializeDataSelector(dsReply));
134 		} else if (r.getNotificationChannel() != null) {
135 			NotificationChannelReply ncReply = r.getNotificationChannel();
136 			writer.write(serializeNotificationChannel(ncReply));
137 		} else if (r.getTrigger() != null) {
138 			TriggerReply trgReply = r.getTrigger();
139 			writer.write(serializeTrigger(trgReply));
140 		} else if (r.getFieldName() != null) {
141 			FieldNameReply fnReply = r.getFieldName();
142 			writer.write(serializeFieldName(fnReply));
143 		} else if (r.getTagField() != null) {
144 			TagFieldReply tfReply = r.getTagField();
145 			writer.write(serializeTagField(tfReply));
146 		} else if (r.getError() != null) {
147 			ErrorType error = r.getError();
148 			writer.write(serializeError(r.getResultCode(), error));
149 		}
150 
151 		writer.write(TERMINATOR);
152 		return writer.toString();
153 	}

155 	public String serialize(Notification n) throws MessageSerializingException {
156 		StringWriter writer = new StringWriter();
157 		writer.write(serializeReplyHeader(n));
158 		if (n.getGoodBye() != null) {
159 			writer.write(GOODBYE);
160 			writer.write(LF);
161 		} else if (n.getReadReport() != null) {
162 			writer.write(serializeNotificationReadReport(n.getReadReport()));
163 		}
164 		// TODO Gibts ReaderInfo in n.getReader() im Text-Format nicht?
165 
166 		return writer.toString();
167 	}

170 	 * Serializes the reply header
171 	 * 
172 	 * @param r
173 	 * @return the reply header
174 	 */
175 	private String serializeReplyHeader(Reply r) {
176 		StringBuffer sBuf = new StringBuffer();
177 		if (r.getId() != null) {
178 			sBuf.append(EXCLAMATION);
179 			sBuf.append(r.getId());
180 			sBuf.append(" ");
181 		}
182 		sBuf.append(OK);
183 		sBuf.append(LF);
184 		return sBuf.toString();
185 	}

241 	 * Converts a calendar into a ISO 8601 formatted String. i.e.
242 	 * YYYY-MM-DDThh:mm:ss.SSSz
243 	 * 
244 	 * @param cal
245 	 *            The Calendar
246 	 * @return String with timestamp formatted as ISO 8601
247 	 */
248 	private String toISO8601(Calendar cal) {
249 		SimpleDateFormat formatter = new SimpleDateFormat();
250 		formatter.applyPattern(ISO_8601_MILLIS_PATTERN);
251 		return formatter.format(cal.getTime());
252 	}

255 		GregorianCalendar gregorianCalendar = new GregorianCalendar(cal.getTimeZone());
256 		gregorianCalendar.setTimeInMillis(cal.getTimeInMillis());
257 		return new XMLGregorianCalendarImpl(gregorianCalendar);
258 	}

261 		Calendar cal = new GregorianCalendar();
262 		cal.set(xmlCal.getYear(), xmlCal.getMonth(), xmlCal.getDay(), xmlCal.getHour(), xmlCal.getMinute(), xmlCal.getSecond());
263 		return cal;
264 	}

348 	/**
349 	 * Serialize a ReadReport used in a reply.
350 	 * 
351 	 * @param writer
352 	 *            The output writer
353 	 * @param report
354 	 *            The ReadReport
355 	 */
356 	private void writeReplyLine(StringWriter writer, ReadReportType report) {
357 
358 		List srcReportList = report.getSourceReport();
359 		Iterator srcReportIt = srcReportList.iterator();
360 		while (srcReportIt.hasNext()) {
361 			SourceReport srcReport = (SourceReport) srcReportIt.next();
362 			String sourceName = null;
363 			if (srcReport.getSourceInfo() != null) {
364 				SourceInfoType sourceInfo = srcReport.getSourceInfo();
365 				sourceName = sourceInfo.getSourceName();
366 			}
367 			List tagList = srcReport.getTag();
368 			Iterator tagListIt = tagList.iterator();
369 			// Add all tags
370 			while (tagListIt.hasNext()) {
371 				/*
372 				 * a Vector of Strings with all attributes that have to be
373 				 * reported
374 				 */
375 				Vector reportAttributes = new Vector();
376 
377 				// --- Add the source name ---
378 				if (sourceName != null) {
379 					reportAttributes.add(sourceName);
380 				}
381 
382 				// --- Add the tag properties ---
383 				TagType tag = (TagType) tagListIt.next();
384 				if (tag.getTagID() != null) {
385 					byte[] tagID = tag.getTagID();
386 					StringBuffer tagIDString = new StringBuffer();
387 					for (int i = 0; i < tagID.length; i++) {
388 						String hexString = Integer.toHexString(new Byte(tagID[i]).intValue()).toString().toUpperCase();
389 						if (hexString.length() == 1) {
390 							tagIDString.append("0");
391 						}
392 						tagIDString.append(hexString);
393 					}
394 					reportAttributes.add(tagIDString.toString());
395 				}
396 				if (tag.getTagIDAsPureURI() != null) {
397 					reportAttributes.add(tag.getTagIDAsPureURI());
398 				}
399 				if (tag.getTagIDAsTagURI() != null) {
400 					reportAttributes.add(tag.getTagIDAsTagURI());
401 				}
402 				if (tag.getTagType() != null) {
403 					reportAttributes.add(tag.getTagType());
404 				}
405 
406 				// --- Add the tag events ---
407 				if (tag.getTagEvent() != null) {
408 					Iterator tagEventIt = tag.getTagEvent().iterator();
409 					while (tagEventIt.hasNext()) {
410 						TagEventType tagEvent = (TagEventType) tagEventIt
411 								.next();
412 						if (tagEvent.getEventType() != null) {
413 							reportAttributes.add(tagEvent.getEventType());
414 						}
415 						if (tagEvent.getTime() != null
416 								&& tagEvent.getTime().getEventTimeTick() != null) {
417 							reportAttributes.add(tagEvent.getTime()
418 									.getEventTimeTick());
419 						} else if (tagEvent.getTime() != null
420 								&& tagEvent.getTime().getEventTimeUTC() != null) {
421 							reportAttributes.add(toISO8601(xmlGregorianCalendarToCalendar(tagEvent.getTime()
422 									.getEventTimeUTC())));
423 						}
424 						// TODO: Should we really report the event triggers???
425 						// if (tagEvent.getEventTriggers() != null &&
426 						// tagEvent.getEventTriggers().getTrigger() != null) {
427 						// //add all the trigger names
428 						// EventTriggersType eventTrigger =
429 						// tagEvent.getEventTriggers();
430 						// Iterator eventTriggerIt =
431 						// eventTrigger.getTrigger().iterator();
432 						// while(eventTriggerIt.hasNext()) {
433 						// String triggerName = (String)eventTriggerIt.next();
434 						// reportAttributes.add(triggerName);
435 						// }
436 						// }
437 
438 					}
439 				}
440 
441 				// --- Add the tag fields ---
442 				if (tag.getTagFields() != null) {
443 					Iterator tagFieldIt = tag.getTagFields().iterator();
444 					while (tagFieldIt.hasNext()) {
445 						TagFieldValueParamType tfvp = (TagFieldValueParamType) tagFieldIt
446 								.next();
447 						// TODO: How to serialize the tagfield
448 						// name-value-pairs??
449 						if (tfvp.getTagFieldName() != null) {
450 							reportAttributes.add(tfvp.getTagFieldName());
451 						}
452 						if (tfvp.getTagFieldValue() != null) {
453 							reportAttributes.add(new String(tfvp
454 									.getTagFieldValue()));
455 						}
456 					}
457 				}
458 
459 				// --- Serialize the tag line ---
460 				writeTagLine(writer, reportAttributes);
461 			}
462 		}
463 	}

941 	 * @param error
942 	 *            The error with description, cause etc.
943 	 * @return
