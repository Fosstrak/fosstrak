670    /**
671     * Removes the specified sources from the list of sources currently
672     * associated with this reader. If one or more of the sources given are not
673     * known, or if some of the sources to be removed are currently not
674     * associated with this reader, these are ignored and all other sources shall
675     * be removed and the command shall complete successfully. But note, in case
676     * of a fixed source an exception ("ERROR_SOURCE_READ_ONLY") is thrown and
677     * none of the sources will be removed
678     * @param sourceList
679     *           The list of sources to remove
680     * @throws ReaderProtocolException
681     *            "ERROR_SOURCE_READ_ONLY"
682     */
683    public final void removeSources(final Source[] sourceList)
684          throws ReaderProtocolException {
685 
686       Vector sources = getVector(sourceList);
687 
688       Enumeration iterator = sources.elements();
689       Source cur;
690       while (iterator.hasMoreElements()) {
691          cur = (Source) iterator.nextElement();
692          if (cur.isFixed()) {
693             throw new ReaderProtocolException("ERROR_SOURCE_READ_ONLY",
694                   MessagingConstants.ERROR_SOURCE_READ_ONLY);
695          }
696       }
697 
698       iterator = sources.elements();
699       while (iterator.hasMoreElements()) {
700          cur = (Source) iterator.nextElement();
701          cur.removeAssociations();
702          this.sources.remove(cur.getName());
703       }
704 
705    }
706 
707    /**
708     * Removes all Sources currently associated with this reader, except the ones
709     * marked as fixed.
710     */
711    public final void removeAllSources() {
712 
713       Enumeration iterator = sources.elements();
714       Source cur;
715 
716       while (iterator.hasMoreElements()) {
717          cur = (Source) iterator.nextElement();
718          if (!cur.isFixed()) {
719             cur.removeAssociations();
720             this.sources.remove(cur.getName());
721          }
722       }
723 
724    }
