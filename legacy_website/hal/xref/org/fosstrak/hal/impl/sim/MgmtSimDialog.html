149 		identifyOnceCheckBox.setSelected(controller.identifyError.contains(readPointName));

151 		writeOnceCheckBox.setSelected(controller.writeError.contains(readPointName));

155 		readContinuousCheckBox.setSelected(controller.continuousReadErrors.contains(readPointName));
156 		writeContinuousCheckBox.setSelected(controller.continuousWriteErrors.contains(readPointName));
157 		killContinuousCheckBox.setSelected(controller.continuousKillErrors.contains(readPointName));

160 	/**
161 	 * Updates the controller to the state of the GUI.
162 	 */
163 	private void updateController() {
164 		if (identifyOnceCheckBox.isSelected() && !controller.identifyError.contains(readPointName)) {
165 			controller.identifyError.add(readPointName);
166 		} else if (!identifyOnceCheckBox.isSelected() && controller.identifyError.contains(readPointName)) {
167 			controller.identifyError.remove(readPointName);
168 		}
169 		
170 		if (readOnceCheckBox.isSelected() && !controller.readError.contains(readPointName)) {
171 			controller.readError.add(readPointName);
172 		} else if (!readOnceCheckBox.isSelected() && controller.readError.contains(readPointName)) {
173 			controller.readError.remove(readPointName);
174 		}
175 		
176 		if (writeOnceCheckBox.isSelected() && !controller.writeError.contains(readPointName)) {
177 			controller.writeError.add(readPointName);
178 		} else if (!writeOnceCheckBox.isSelected() && controller.writeError.contains(readPointName)) {
179 			controller.writeError.remove(readPointName);
180 		}
181 		
182 		if (killOnceCheckBox.isSelected() && !controller.killError.contains(readPointName)) {
183 			controller.killError.add(readPointName);
184 		} else if (!killOnceCheckBox.isSelected() && controller.killError.contains(readPointName)) {
185 			controller.killError.remove(readPointName);
186 		}
187 		
188 		if (identifyContinuousCheckBox.isSelected() && !controller.continuousIdentifyErrors.contains(readPointName)) {
189 			controller.continuousIdentifyErrors.add(readPointName);
190 		} else if (!identifyContinuousCheckBox.isSelected() && controller.continuousIdentifyErrors.contains(readPointName)) {
191 			controller.continuousIdentifyErrors.remove(readPointName);
192 		}
193 		
194 		if (readContinuousCheckBox.isSelected() && !controller.continuousReadErrors.contains(readPointName)) {
195 			controller.continuousReadErrors.add(readPointName);
196 		} else if (!readContinuousCheckBox.isSelected() && controller.continuousReadErrors.contains(readPointName)) {
197 			controller.continuousReadErrors.remove(readPointName);
198 		}
199 		
200 		if (writeContinuousCheckBox.isSelected() && !controller.continuousWriteErrors.contains(readPointName)) {
201 			controller.continuousWriteErrors.add(readPointName);
202 		} else if (!writeContinuousCheckBox.isSelected() && controller.continuousWriteErrors.contains(readPointName)) {
